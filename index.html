import { useEffect } from 'react'; // All website + admin CSS in one place const GlobalCSS = () => { useEffect(()=>{ if(!document.querySelector('meta[name="viewport"]')){ const m=document.createElement('meta'); m.name='viewport'; m.content='width=device-width,initial-scale=1,viewport-fit=cover'; document.head.prepend(m); } },[]); return (); }; export { GlobalCSS };
